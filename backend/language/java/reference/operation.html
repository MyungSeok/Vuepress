  String color;

  void driveSedan() {

  }

class SportsCar extends Car {
  void driveSportCar() {
    System.out.println("Start Sports Car");
  }
}

Sedan sedan = new Sedan();
SportsCar sportsCar = new SportsCar();

isCar(sedan);
isCar(sportsCar);

void isCar(Car car) {
  if (car instanceof Sedan) {
    Sedan s = (Sedan) car;
    s.driveSedan();
  } else if (car instanceof SportsCar) {
    SportsCar p = (SportsCar) car;
    p.driveSportCar();
  }
}
